### Tailwind Classes (Recommended)

<CodeBlock
  language="tsx"
  code={`// Text colors
<p className="text-primary">Main text</p>
<p className="text-secondary">Muted text</p>
<p className="text-error">Error message</p>

// Background colors
<div className="bg-surface">Card</div>
<div className="bg-accent-bg">Accent highlight</div>

// Combined patterns
<button className="bg-accent-strong text-inverse hover:bg-accent-strong/90">
  Primary Button
</button>

<div className="bg-error-light text-error border-error rounded-md p-4">
  Error alert message
</div>`}
/>

### TypeScript Tokens (Dynamic Styles)

<CodeBlock
  language="typescript"
  code={`import { ALIAS, ABYSS, DEEP_CURRENT } from '@adrozdenko/design-system'

// Use ALIAS tokens for semantic meaning
const styles = {
  background: ALIAS.background.surface,
  color: ALIAS.text.primary,
  borderColor: ALIAS.border.default,
}

// Dynamic state
const bgColor = isError
  ? ALIAS.status.error
  : ALIAS.background.surface`}
/>

<h2 style={{ marginTop: SPACING.px.sectionHeadingTop, marginBottom: SPACING.px.sectionHeadingBottom }}>Guidelines</h2>

<div style={{
  display: 'grid',
  gridTemplateColumns: 'repeat(2, 1fr)',
  gap: SPACING.px.cardGap,
  marginBottom: SPACING.px.section,
}}>
  <GuidelineCard type="do" title="Use ALIAS tokens for components">
    Always use semantic tokens like `ALIAS.text.primary` instead of raw values. This ensures consistency and easy theming.
  </GuidelineCard>

  <GuidelineCard type="dont" title="Hardcode hex values">
    Never use raw hex values like `#1D3461` in components. Use the token system for maintainability.
  </GuidelineCard>

  <GuidelineCard type="do" title="Use Tailwind classes for static styles">
    For static color usage, prefer Tailwind classes like `text-primary` over inline styles.
  </GuidelineCard>

  <GuidelineCard type="dont" title="Use default Tailwind colors">
    Don't use Tailwind's default palette (blue-500, gray-200). Use DDS semantic tokens instead.
  </GuidelineCard>
</div>
