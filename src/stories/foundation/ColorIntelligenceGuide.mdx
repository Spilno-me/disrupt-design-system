→ Returns: bg, text, border, shadow, glass options

# 3. If using custom colors, validate contrast
mcp__dds__check_contrast({

mcp__dds__get_accessible_colors({

  minLevel: "AAA"

<h2 style={{ marginTop: SPACING.px.sectionHeadingTop, marginBottom: SPACING.px.sectionHeadingBottom }}>Quick Reference</h2>

<div style={{
  display: 'grid',
  gridTemplateColumns: 'repeat(2, 1fr)',
  gap: SPACING.px.cardGap,
  marginBottom: SPACING.px.section,
}}>
  <GuidelineCard type="do" title="Use Context Queries">
    Always start with <code>get_color_recommendation(context)</code> instead of picking colors manually.
    The system knows which colors work together.
  </GuidelineCard>

  <GuidelineCard type="dont" title="Hardcode Hex Values">
    Never use <code>#08A4BD</code> directly. Use semantic tokens like <code>text-accent</code> that adapt to themes.
  </GuidelineCard>

  <GuidelineCard type="do" title="Validate Badge Contrast">
    Badges use small text (12-14px) which requires AAA (7:1 ratio), not just AA (4.5:1).
    Always verify with <code>check_contrast</code>.
  </GuidelineCard>

  <GuidelineCard type="dont" title="Same Family on Itself">
    Never put ABYSS text on ABYSS background, or CORAL on CORAL. Same color family = invisible text.
  </GuidelineCard>
</div>
